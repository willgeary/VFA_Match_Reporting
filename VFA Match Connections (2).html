fig, ax = plt.subplots(figsize=(10,10))

x = range(len(df))
y = np.array(df['4/13/2016'])
z = np.array(df['4/20/2016'])

april_thirteen = ax.scatter(x, y, color='gray')
april_twenty = ax.scatter(x, z, color='darkblue')
ax.axhline(10,color='gray', linestyle='dashed', lw=2)
ax.axhline(14,color='darkblue', linestyle='dashed', lw=2)
ax.set_xlabel('(each dot represents a fellow)', fontsize=16)
ax.set_ylabel('# of Connections', fontsize=16)
ax.tick_params(labelsize=12)
ax.set_title('Total # of Connections per Fellow', fontsize=20)
ax.tick_params(labelbottom='off')
ax.annotate('Average on 4/13/15= \n10 Connections', xy=(135,7.5), color='gray', fontsize=14)
ax.annotate('Average on 4/20/15= \n14 Connections', xy=(135,11.5),color='darkblue', fontsize=14)
ax.annotate('Max = 35 Connections', xy=(105,34.7),color='black', fontsize=16)
ax.annotate('$\}$',xy=(28, 1), fontsize=50, color='red')
ax.annotate('Less than five: let\'s giddy up!', xy=(38,1.8), color='red', fontsize=18)
ax.set_xlim([-20,len(x)+10])
ax.set_ylim([0,38])

ax.legend([april_twenty, april_thirteen], ["As of 4/20/16", "As of 4/13/16"], loc=2, fontsize=16);

fig.savefig('fig.jpg', dpi=250)
